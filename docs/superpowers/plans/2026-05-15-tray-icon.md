# Tray Icon Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Replace CommandDeck's invisible tray icon with a stateful 2×2 grid icon that reflects live process count and surfaces crash/unexpected-exit alerts via a coloured badge dot.

**Architecture:** A new `src/tray-icon.js` module owns all icon generation — a pure `buildTrayIconSvg()` function (testable with plain Node) and a `buildTrayIcon()` that wraps it with `nativeImage`. `main.js` gains an `alertState` variable, a `killedByUser` Set (tracks PIDs the user intentionally killed, separate from `liveProcesses` to avoid a deletion-order race), and an `updateTrayIcon()` function called on every state-changing event.

**Tech Stack:** Electron v29+ `nativeImage`, SVG data URLs, Node.js built-in `node:test` runner (no new npm packages).

## Task 2: Implement `buildTrayIconSvg()` — Make Tests Pass

**Files:**

- [ ] **Step 1: Create `src/tray-icon.js` with the pure SVG generator**

```js
// Fill order: top-left → bottom-right → top-right → bottom-left
// Diagonal pattern reads clearly at 22px (avoids L-shape)
const FILL_ORDER = [
  { x: 2,  y: 2  },  // top-left
  { x: 12, y: 12 },  // bottom-right
  { x: 12, y: 2  },  // top-right
  { x: 2,  y: 12 },  // bottom-left
];

const CELL = 8;
const CORNER = 1.5;

const COLOR = {
  bg:        '#0c0e14',
  active:    '#4ade80',  // Linux / Windows
  activeMac: '#000000',  // macOS template image
  idle:      '#3a4060',
  red:       '#f87171',
  amber:     '#fbbf24',
};

function buildTrayIconSvg(runningCount, alertLevel, platform, size = 22) {
  const activeColor = platform === 'darwin' ? COLOR.activeMac : COLOR.active;
  const filled = Math.min(runningCount, 4);

  const cells = FILL_ORDER.map((cell, i) =>
    i < filled
      ? `<rect x="${cell.x}" y="${cell.y}" width="${CELL}" height="${CELL}" rx="${CORNER}" fill="${activeColor}"/>`
      : `<rect x="${cell.x}" y="${cell.y}" width="${CELL}" height="${CELL}" rx="${CORNER}" fill="none" stroke="${COLOR.idle}" stroke-width="1"/>`
  ).join('');

  const badgeColor = alertLevel === 'red' ? COLOR.red : alertLevel === 'amber' ? COLOR.amber : null;
  const badge = badgeColor ? `<circle cx="19" cy="3" r="3" fill="${badgeColor}"/>` : '';

  return `<svg width="${size}" height="${size}" viewBox="0 0 22 22" xmlns="http://www.w3.org/2000/svg"><rect width="22" height="22" rx="3" fill="${COLOR.bg}"/>${cells}${badge}</svg>`;
}

function buildTrayIcon(runningCount, alertLevel) {
  // Lazy require — keeps this file importable by plain Node (for tests)
  const { nativeImage } = require('electron');
  const platform = process.platform;
  const size = platform === 'linux' ? 22 : 16;

  const svg1x = buildTrayIconSvg(runningCount, alertLevel, platform, size);
  const url1x = `data:image/svg+xml;base64,${Buffer.from(svg1x).toString('base64')}`;
  const img = nativeImage.createFromDataURL(url1x);

  // Add 2x representation for Retina / HiDPI (macOS and Windows)
  if (platform === 'darwin' || platform === 'win32') {
    const svg2x = buildTrayIconSvg(runningCount, alertLevel, platform, size * 2);
    const url2x = `data:image/svg+xml;base64,${Buffer.from(svg2x).toString('base64')}`;
    img.addRepresentation({ scaleFactor: 2.0, dataURL: url2x });
  }

  if (platform === 'darwin') img.setTemplateImage(true);

  return img;
}

module.exports = { buildTrayIconSvg, buildTrayIcon };
```

This script launches a hidden Electron BrowserWindow, renders each icon state on a canvas, and prints base64 PNG data URLs to stdout.

```js
const { app, BrowserWindow } = require('electron');

app.whenReady().then(async () => {
  const win = new BrowserWindow({
    show: false,
    width: 100,
    height: 100,
    webPreferences: { nodeIntegration: true, contextIsolation: false },
  });

  const states = [
    { count: 0, alert: null,    platform: 'linux',  size: 22, name: 'idle-linux' },
    { count: 2, alert: null,    platform: 'linux',  size: 22, name: 'active-linux' },
    { count: 4, alert: null,    platform: 'linux',  size: 22, name: 'full-linux' },
    { count: 2, alert: 'red',   platform: 'linux',  size: 22, name: 'alert-red-linux' },
    { count: 2, alert: 'amber', platform: 'linux',  size: 22, name: 'alert-amber-linux' },
    { count: 0, alert: null,    platform: 'darwin', size: 16, name: 'idle-mac' },
    { count: 2, alert: null,    platform: 'darwin', size: 16, name: 'active-mac' },
    { count: 4, alert: null,    platform: 'darwin', size: 16, name: 'full-mac' },
    { count: 2, alert: 'red',   platform: 'darwin', size: 16, name: 'alert-red-mac' },
    { count: 2, alert: 'amber', platform: 'darwin', size: 16, name: 'alert-amber-mac' },
  ];

  const { buildTrayIconSvg } = require('./src/tray-icon');
  const results = {};

  for (const s of states) {
    const svg = buildTrayIconSvg(s.count, s.alert, s.platform, s.size);
    const dataUrl = await win.webContents.executeJavaScript(`
      new Promise(resolve => {
        const img = new Image();
        img.onload = () => {
          const c = document.createElement('canvas');
          c.width = ${s.size}; c.height = ${s.size};
          c.getContext('2d').drawImage(img, 0, 0);
          resolve(c.toDataURL('image/png'));
        };
        img.src = 'data:image/svg+xml;base64,${Buffer.from(svg).toString('base64')}';
      })
    `);
    results[s.name] = dataUrl;
  }

  console.log(JSON.stringify(results, null, 2));
  app.quit();
});
```

function buildTrayIcon(runningCount, alertLevel) {
  const { nativeImage } = require('electron');
  const ICONS = require('./tray-icons-data');
  const platform = process.platform;

  const baseState = runningCount === 0 ? 'idle' : runningCount >= 4 ? 'full' : 'active';
  const alertSuffix = alertLevel ? `-${alertLevel}` : '';
  const platformSuffix = platform === 'darwin' ? 'mac' : 'linux';
  const key = `${baseState}${alertSuffix}-${platformSuffix}`;

}
